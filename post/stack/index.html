实现一个简单的栈:

package main

import (
	"fmt"
	"testing"
)

type Stack []int

func (s *Stack) Push(val int) {
	*s = append(*s, val)
}

func (s *Stack) Pop() int {
	if len(*s) == 0 {
		return -1
	}

	val := (*s)[len(*s)-1]
	*s = (*s)[:len(*s)-1]
	return val
}

func (s *Stack) Peek() int {
	if len(*s) == 0 {
		return -1
	}

	return (*s)[len(*s)-1]
}

func (s *Stack) IsEmpty() bool {
	return len(*s) == 0
}

func TestStack(t *testing.T) {
	s := Stack{}
	fmt.Println(s.IsEmpty()) // true

	s.Push(1)
	s.Push(3)
	s.Push(5)
	fmt.Println(s) // [1 3 5]

	fmt.Println(s.Peek()) // 5

	fmt.Println(s.Pop()) // 5
	fmt.Println(s)       // [1 3]
}
